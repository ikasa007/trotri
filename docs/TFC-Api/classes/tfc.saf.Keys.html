Keys

Keys class file 密钥管理类

配置 /cfg/key/cluster.php：

return array (
  'authentication' => array (
    'crypt' => string,   // 加密密钥
    'sign' => string,    // 签名密钥
    'expiry' => integer, // 缺省的密文有效期，如果等于0，表示永久有效，单位：秒
    'rnd_len' => integer // 随机密钥长度，取值 0-32
  ),
  'site' => array (
    'crypt' => string,   // 加密密钥
    'sign' => string,    // 签名密钥
    'expiry' => integer, // 缺省的密文有效期，如果等于0，表示永久有效，单位：秒
    'rnd_len' => integer // 随机密钥长度，取值 0-32
  )
);
